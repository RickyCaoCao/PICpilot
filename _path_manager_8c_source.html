423  heading = deg2rad(90 - heading);
424 
425 
426  float orbitDistance = sqrt(pow(position[0] - center[0],2) + pow(position[1] - center[1],2));
427  float courseAngle = atan2(position[1] - center[1], position[0] - center[0]); // (y,x) format
428 
429  while (courseAngle - heading < -PI){
430  courseAngle += 2 * PI;
431  }
432  while (courseAngle - heading > PI){
433  courseAngle -= 2 * PI;
434  }
435 
436  return 90 - rad2deg(courseAngle + direction * (PI/2 + atan(k_gain[ORBIT] * (orbitDistance - radius)/radius))); //Heading in degrees (magnetic)
437 }
