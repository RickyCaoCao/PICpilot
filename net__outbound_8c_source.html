184 void sendNextByte(void) {
185  unsigned char sendByte; // The byte to send
186  if ( stagingBuffer.sendIndex < API_HEADER_LENGTH ) {
187  //while (U2STAbits.TRMT == 0);
188  sendByte = stagingBuffer.header[stagingBuffer.sendIndex] & 0xFF;
189  // Compute checksum
190  if (stagingBuffer.sendIndex >= 3) {
191  stagingBuffer.checksum += sendByte & 0xFF;
192  }
193  } else if ( stagingBuffer.sendIndex < PACKET_LENGTH - 1 ) {
194  sendByte = stagingBuffer.telemetry.asArray[stagingBuffer.sendIndex - API_HEADER_LENGTH] & 0xFF;
195  stagingBuffer.checksum += sendByte & 0xFF;
196  } else if ( stagingBuffer.sendIndex == PACKET_LENGTH - 1) {
197  sendByte = 0xFF - (stagingBuffer.checksum & 0xFF);
198  } else {
199  IFS1bits.U2TXIF = 0;
200  return;
201  }
202 
203  stagingBuffer.sendIndex++;
204  IFS1bits.U2TXIF = 0;
205  U2TXREG = sendByte;
206 }

232  unsigned int apiIndex = 0;

269 // Send a telemetry block immediately (blocking)
270 int sendTelemetryBlock(struct telem_block *telem) {
271 
272  unsigned char apiHeader[API_HEADER_LENGTH];
273  unsigned int headerLength = generateApiHeader(apiHeader, 0);
274  // Treat the telemetry block as a character array.
275  unsigned char *telemAsArray = (unsigned char*)telem;
276 
277  unsigned int lengthCheck = 0;
278  unsigned char checksum = 0;
279  unsigned int i;
280  // Send the header
281  for (i = 0; i < headerLength; i++ ) {
282  // Culmulative checksum
283  if (i >= API_HEADER_PREFIX) {
284  checksum += apiHeader[i] & 0xFF;
285  }
286  // Wait for data link to clear from previous send
287  while (U2STAbits.TRMT == 0);
288  // Load transmit register
289  U2TXREG = apiHeader[i];
290  }
291  lengthCheck += i;
292  // Send the telemetry
293  for (i = 0; i < sizeof(struct telem_block); i++) {
294  // Culmulative checksum
295  checksum += telemAsArray[i] & 0xFF;
296  // Wait for data link to clear from previous send
297  while (U2STAbits.TRMT == 0);
298  // Load transmit register
299  U2TXREG = telemAsArray[i];
300  }
301  lengthCheck += i;
302  // Send the checksum
303  while (U2STAbits.TRMT == 0);
304  U2TXREG = 0xFF - checksum;
305  lengthCheck += 1;
306 
307  // Note: We send the last piece of data through UART and
308  // then forget about it. We assume it will get handled eventually
309  return 0;
310 }
